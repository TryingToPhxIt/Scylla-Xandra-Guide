lib/scylla/application.exs

  def start(_type, _args) do
    xandra_config = Application.fetch_env!(:your_app, :xandra_repo)

    children = [
      ...
      {Xandra, Keyword.put(xandra_config, :name, :scylla_db)},
      ...
    ]
  
    opts = [strategy: :one_for_one, name: Your_APP.Supervisor]
    Supervisor.start_link(children, opts)
  end

  Note: :scylla_db is a named value. It is used when executing Xandra queries.
